Code Examples

See the framework in action with real code samples

import { BasePage } from './core/BasePage';
import type { ILocator, IUserCredentials } from './core/interfaces';

export class LoginPage extends BasePage {
    private readonly locators = {
        usernameInput: {
            strategy: 'testId' as const,
            value: 'username-input',
            description: 'Username input field',
        },
        passwordInput: {
            strategy: 'testId' as const,
            value: 'password-input',
            description: 'Password input field',
        },
        loginButton: {
            strategy: 'testId' as const,
            value: 'login-btn',
            description: 'Login submit button',
        },
    };

    public async login(credentials: IUserCredentials): Promise<void> {
        await this.fill(this.locators.usernameInput, credentials.username);
        await this.fill(this.locators.passwordInput, credentials.password);
        await this.click(this.locators.loginButton);
    }
}
